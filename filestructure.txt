Key Design Decisions
1. User identification — The frontend should send user_id via an X-User-ID header (or later, a JWT token). A FastAPI dependency extracts and validates it.

2. Chatbot instance management — MultiLayerChatbot does Postgres schema setup in __init__, so we cache instances per user_id using a dict (or functools.lru_cache). This avoids re-running setup on every request.

3. CORS — The frontend team will need CORS enabled. Add configurable CORS_ORIGINS to settings.

4. No changes to core logic — All routes are thin wrappers around MultiLayerChatbot methods. The memory pipeline, LangGraph, Pinecone, etc. are untouched.

5. One print() to remove — prune_stale_memories() has a print() that would leak into server logs instead of the response. Return the count only.

app/api/dependencies.py — Chatbot DI

from __future__ import annotations  
from typing import Dict  
from fastapi import Header, HTTPException  
from app.chatbot import MultiLayerChatbot  
from app.security.isolation import validate_user_id, SecurityError  
  
_chatbot_cache: Dict[str, MultiLayerChatbot] = {}  
  
def get_chatbot(x_user_id: str = Header(..., alias="X-User-ID")) -> MultiLayerChatbot:  
    """  
    FastAPI dependency: extract user_id from header, return cached chatbot.  
    """  
    try:  
        user_id = validate_user_id(x_user_id)  
    except SecurityError as e:  
        raise HTTPException(status_code=400, detail=str(e))  
  
    if user_id not in _chatbot_cache:  
        try:  
            _chatbot_cache[user_id] = MultiLayerChatbot(user_id=user_id)  
        except Exception as e:  
            raise HTTPException(status_code=503, detail=f"Failed to init chatbot: {e}")  
  
    return _chatbot_cache[user_id]
app/api/routes/chat.py

from fastapi import APIRouter, Depends, HTTPException  
from app.api.schemas import ChatRequest, ChatResponse  
from app.api.dependencies import get_chatbot  
from app.chatbot import MultiLayerChatbot  
  
router = APIRouter(prefix="/api", tags=["chat"])  
  
@router.post("/chat", response_model=ChatResponse)  
def chat(req: ChatRequest, chatbot: MultiLayerChatbot = Depends(get_chatbot)):  
    try:  
        response = chatbot.chat(req.message)  
        return ChatResponse(response=response, user_id=chatbot.user_id)  
    except Exception as e:  
        raise HTTPException(status_code=500, detail=str(e))
app/api/routes/memories.py

from fastapi import APIRouter, Depends, HTTPException, Query  
from typing import Optional  
from app.api.schemas import MemoriesResponse, SearchResponse, HistoryMessage  
from app.api.dependencies import get_chatbot  
from app.chatbot import MultiLayerChatbot  
  
router = APIRouter(prefix="/api", tags=["memories"])  
  
@router.get("/memories", response_model=MemoriesResponse)  
def get_memories(chatbot: MultiLayerChatbot = Depends(get_chatbot)):  
    return chatbot.get_memories()  
  
@router.get("/memories/search", response_model=SearchResponse)  
def search_memories(  
    q: str = Query(..., min_length=1),  
    top_k: int = Query(10, ge=1, le=100),  
    category: Optional[str] = None,  
    chatbot: MultiLayerChatbot = Depends(get_chatbot),  
):  
    results = chatbot.search_memories(q, top_k=top_k, category=category)  
    return SearchResponse(query=q, results=results)  
  
@router.get("/history", response_model=list[HistoryMessage])  
def get_history(chatbot: MultiLayerChatbot = Depends(get_chatbot)):  
    return chatbot.get_conversation_history()  
  
@router.delete("/memories/{memory_id}")  
def delete_memory(memory_id: str, chatbot: MultiLayerChatbot = Depends(get_chatbot)):  
    try:  
        chatbot.delete_memory(memory_id)  
        return {"status": "deleted", "memory_id": memory_id}  
    except Exception as e:  
        raise HTTPException(status_code=404, detail=str(e))  
  
@router.delete("/memories")  
def delete_all_memories(chatbot: MultiLayerChatbot = Depends(get_chatbot)):  
    chatbot.delete_all_memories()  
    return {"status": "all_deleted", "user_id": chatbot.user_id}
